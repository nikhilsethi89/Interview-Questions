# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def binaryTreePaths(self, root):
        """
        :type root: TreeNode
        :rtype: List[str]
        """
        if root==None:
            return []
        path=""
        result=[]
        
        self.getpath(path,root,result)
           
        return result
    
    def getpath(self,path,node,result):
        if node.left==None and node.right==None:
            return result.append(path+str(node.val))
        if node.left!=None:
            self.getpath(path+str(node.val)+"->",node.left,result)
        if node.right!=None:
            self.getpath(path+str(node.val)+"->",node.right,result)
