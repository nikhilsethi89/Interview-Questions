# The isBadVersion API is already defined for you.
# @param version, an integer
# @return a bool
# def isBadVersion(version):

class Solution(object):
    def firstBadVersion(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n==0:
            return 0
        
        min=1
        max=n
        
        while (min!=max):
            
            half=(min+max)/2
            if isBadVersion(half)==True:
                max=half
            else:
                min=half+1
        
        return min
